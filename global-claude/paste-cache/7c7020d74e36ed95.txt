# Name: Release Deploy (Production versions.)
# Description: Tag, build, and publish a Docker image to ACR on PR-merge pushes to release/deploy.
#

name: Release Deploy

on:
    push:
        # Only run on pushes to the release/deploy branch
        branches:
            - release/production

permissions:
    contents: write

jobs:
    deploy_on_staging:
        # Only run when the push is a merge commit from a PR
        if: ${{ github.event.head_commit && contains(github.event.head_commit.message, 'Merge pull request') }}
        runs-on: ubuntu-latest

        # Use the environment-scoped secrets/vars from "development"
        environment: development

        env:
            # ACR login server, e.g. myregistry.azurecr.io
            REGISTRY: ${{ vars.AZ_REGISTER_CONTAINER }}

            # Image name derived from the repository name
            IMAGE_NAME: ${{ vars.IMAGE_NAME }}

        # Steps to execute in the job
        steps:

            # 1) Login to Azure Container Registry
            - name: Login to Azure Container Registry
              uses: azure/docker-login@v2
              with:
                  # Uses environment-scoped variable and secrets
                  login-server: ${{ vars.AZ_REGISTER_CONTAINER }}
                  username: ${{ secrets.AZ_CLIENT_ID }}
                  password: ${{ secrets.AZ_CLIENT_SECRET }}

            # 2) Pull the latest dev image, retag as latest, and push
            - name: Retag dev-latest to latest
              shell: bash
              run: |
                  set -euo pipefail
                  docker pull "${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:dev-latest"
                  docker tag "${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:dev-latest" "${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:latest"
                  docker push "${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:latest"
            
            # 3): Authenticate with Azure CLI ───────────────────────────────────
            # Logs into Azure using a service principal. This gets authorization to modify
            # cloud infrastructure directly (such as Container Apps).
            - name: Azure Login
              uses: azure/login@v2
              with:
                creds: '{"clientId":"${{ secrets.AZ_CLIENT_ID }}","clientSecret":"${{ secrets.AZ_CLIENT_SECRET }}","subscriptionId":"${{ secrets.AZ_SUBSCRIPTION_ID }}","tenantId":"${{ secrets.AZ_TENANT_ID }}"}'

            # ── STEP 8: Deploy to Azure Container App ─────────────────────────────────
            # Instructs the designated Azure Container App to pull and run the newly built
            # 'dev-latest' image, ensuring the live environments run the latest changes.
            - name: Update Azure Container App
              uses: azure/CLI@v2
              with:
                inlineScript: |
                  az containerapp update \
                    --name ${{ vars.AZ_CONTAINER_APP_NAME_PROD }} \
                    --resource-group ${{ vars.AZ_RESOURCE_GROUP_PROD }} \
                    --image ${{ env.REGISTRY }}/${{ env.IMAGE_NAME }}:latest
